git add 提交到暂存区
git commit -m"提交"  提交到历史区
git commit -a -m "提交" 直接提交到历史区
git checkout 分支名 切换分支
git checkout -b 分支名 创建并切换到分支名
git branch 查询分支
git branch 分支名 新建分支
git merge 合并分支
touch 新建一个文件
git diff 对比
git reset --hard
git branch -D 分知名 删除分支（不能在本分支删自己）
rm 文件名 删除文件
rm -rf 文件夹名 删除文件夹
git status 是否有文件被修改过
git diff文件名 查询被修改的内容
git log 查询提交历史
